## My Python3 Solution

```python
def findMedianOfSortedArray(sortedArray: List[int]) -> float:
    if len(sortedArray) % 2 == 1:
        index = int(((len(sortedArray)+1)/2)-1)
        return float(sortedArray[index])
    else:
        index = int((len(sortedArray)/2)-1)
        result = ((sortedArray[index] + sortedArray[index+1])/2)
        return result

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        sortedList = sorted(nums1+nums2)
        result = findMedianOfSortedArray(sortedList)
        return result
```
